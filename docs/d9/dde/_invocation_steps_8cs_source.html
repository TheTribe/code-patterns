    26 using System;
    27 
    28 using Castle.DynamicProxy;
    29 
    30 using Moq;
    31 
    32 using Patterns.Specifications.Models.Interception;
    33 using Patterns.Specifications.Models.Mocking;
    34 
    35 using TechTalk.SpecFlow;
    36 
    37 namespace Patterns.Specifications.Steps.Interception
    38 {
    39  [Binding]
    40  public class InvocationSteps
    41  {
    42  private readonly InterceptionContext _context;
    43  private readonly MoqContext _moq;
    44 
    45  public InvocationSteps(InterceptionContext context, MoqContext moq)
    46  {
    47  _context = context;
    48  _moq = moq;
    49  }
    50 
    51  [Given(@"I have a mocked invocation")]
    52  public void CreateMockedInvocation()
    53  {
    54  Mock<IInvocation> mockCall = _moq.Container.Mock<IInvocation>();
    55  SetInvocationToObjectToString(mockCall);
    56  _context.Invocation = mockCall.Object;
    57  }
    58 
    59  [Given(@"I have a mocked invocation that returns a value")]
    60  public void CreateMockedInvocationWithReturn()
    61  {
    62  Mock<IInvocation> mockCall = _moq.Container.Mock<IInvocation>();
    63  SetInvocationToObjectToString(mockCall);
    64  mockCall.SetupGet(call => call.ReturnValue).Returns("hello world");
    65  _context.Invocation = mockCall.Object;
    66  }
    67 
    68  [Given(@"I have a mocked invocation that throws an Exception")]
    69  public void CreateMockedInvocationWithException()
    70  {
    71  Mock<IInvocation> mockCall = _moq.Container.Mock<IInvocation>();
    72  SetInvocationToObjectToString(mockCall);
    73  mockCall.Setup(call => call.Proceed()).Throws(new Exception());
    74  _context.Invocation = mockCall.Object;
    75  }
    76 
    77  [Then(@"the mocked invocation should have been instructed to proceed")]
    78  public void VerifyMockInvocationProceed()
    79  {
    80  _moq.Container.Mock<IInvocation>().Verify(call => call.Proceed(), Times.Once());
    81  }
    82 
    83  private static void SetInvocationToObjectToString(Mock<IInvocation> mockCall)
    84  {
    85  mockCall.SetupGet(call => call.TargetType).Returns(typeof (object));
    86  mockCall.SetupGet(call => call.Method).Returns(typeof (object).GetMethod("ToString"));
    87  }
    88  }
    89 }
