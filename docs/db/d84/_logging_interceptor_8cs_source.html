    60  public void Intercept(IInvocation invocation)
    61  {
    62  ILog log = _logFactory(invocation.TargetType);
    63  log.Trace(handler => handler(LoggingResources.MethodStartTraceFormat, invocation.TargetType, invocation.Method.Name));
    64  log.Debug(handler => handler(LoggingResources.MethodArgsDebugFormat, invocation.Method.Name, GetMethodArguments(invocation)));
    65 
    66  try
    67  {
    68  invocation.Proceed();
    69  log.Info(handler => handler(LoggingResources.MethodInfoFormat, invocation.Method.Name, LoggingResources.MethodInfoPass));
    70  }
    71  catch (Exception error)
    72  {
    73  log.Info(handler => handler(LoggingResources.MethodInfoFormat, invocation.Method.Name, LoggingResources.MethodInfoFail));
    74  log.Error(handler => handler(LoggingResources.ExceptionErrorFormat, invocation.Method.Name, error.ToFullString()));
    75  if (!_config.TrapExceptions) throw;
    76  }
    77 
    78  log.Trace(handler => handler(LoggingResources.MethodStopTraceFormat, invocation.TargetType, invocation.Method.Name));
    79 
    80  if (invocation.ReturnValue == null) return;
    81 
    82  object value = ConvertValueForDisplay(invocation.ReturnValue);
    83  log.Debug(handler => handler(LoggingResources.MethodReturnDebugFormat, invocation.Method.Name, value));
    84  }
