using System;
using System.Text;

namespace Patterns.Logging
{
	/// <summary>
	///    Provides type mixins designed to aid in logging operations.
	/// </summary>
	public static class Mixins
	{
		/// <summary>
		///    Creates an aggregated exception message that includes
		///    messages from all inner exceptions.
		/// </summary>
		/// <param name="error">The error.</param>
		/// <returns>The aggregated exception message.</returns>
		public static string ToFullString(this Exception error)
		{
			StringBuilder builder = new StringBuilder(error.Message)
				.AppendLine()
				.AppendLine(error.StackTrace);

			if (error.InnerException != null) builder.AppendFormat(LoggingResources.InnerExceptionFormat, error.InnerException.ToFullString());

			return builder.ToString();
		}
	}
}
