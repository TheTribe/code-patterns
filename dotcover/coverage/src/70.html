using Patterns.Specifications.Models;

using TechTalk.SpecFlow;

namespace Patterns.Specifications.Steps
{
	[Binding]
	public class ObjectSteps
	{
		private readonly ObjectContext _context;

		public ObjectSteps(ObjectContext context)
		{
			_context = context;
		}

		[Given(@"my object is an integer with the value (.*)")]
		public void AssignIntegerTarget(int value)
		{
			_context.Target = value;
		}
	}
}
